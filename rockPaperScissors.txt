function randomInt() {
	let min = 0; 
    	let max = 3;  
    	let result = Math.floor(Math.random() * (+max - +min)) + +min;
	return result;
}

function computerSelection() {
	let choice = randomInt();
	switch (choice) {
		case 0: move = "rock";
		break;
		case 1: move = "paper";
		break;
		case 2: move = "scissors";
		break;
	}
	return move;
}

function playerSelection() {
	let choice = prompt("Rock, paper, or scissors?", "");
	let usableChoice = choice.toLowerCase();
	return usableChoice;
}

function round(scores) {
	let playerMove = playerSelection();
	let computerMove = computerSelection();
	if (playerMove == "rock" && computerMove == "paper")
		{alert("Computer used paper! Computer wins!");
		scores[1]++}
	else if (playerMove == "rock" && computerMove == "scissors")
		{alert("Computer used scissors! Player wins!");
		scores[0]++}
	else if (playerMove == "scissors" && computerMove == "rock")
		{alert("Computer used rock! Computer wins.");
		scores[1]++}
	else if (playerMove == "scissors" && computerMove == "paper")
		{alert("Computer used paper! Player wins!");
		scores[0]++}
	else if (playerMove == "paper" && computerMove == "scissors")
		{alert("Computer used scissors! Computer wins.");
		scores[1]++}
	else if (playerMove == "paper" && computerMove == "rock")
		{alert("Computer used rock! Player wins!");	
		scores[0]++}
	else if (playerMove == "gun")
		{alert("Whoa, put down the gun! You win!");
		scores[0] = 5;}
	else {alert(`Computer used ${computerMove}! It\'s a tie!`)};
	alert(`Player\'s score is: ${scores[0]}. Computer\'s score is: ${scores[1]}.`);
	return scores;
}

function game() {
	let playerScore = 0;
	let computerScore = 0;
	let scores = [playerScore, computerScore];
	round(scores);
	
	round(scores);
	
	round(scores);
	
	round(scores);

	if (scores[0] > scores[1]) {alert("PLAYER BEAT THE COMPUTER!")}

}
